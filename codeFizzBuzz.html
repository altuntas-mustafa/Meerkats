export function fizzBuzzKata(numArr:number[]): string[]{
const newNumArray = numArr.map(checkFizzBuzz);
return newNumArray;
}

function checkFizzBuzz (number: number) : string {
switch(true) {
case (number < 1 || number > 100):
return 'Please write a number between 1 and 100';
case (number % 3 == 0 && number % 5 == 0):
return "fizzbuzz";
case (number % 3 == 0):
return "fizz";
case (number % 5 == 0):
return "buzz";
default:
return number.toString();
}
}
